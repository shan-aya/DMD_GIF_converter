# Ajouter ces méthodes dans DMDConverter
    def load_translations(self):
        """Charge les traductions"""
        lang = self.lang.get()
        try:
            with open(f'lang_{lang}.json', 'r', encoding='utf-8') as f:
                return json.load(f)
        except:
            return {}
    
    def t(self, key):
        """Traduit une clé"""
        return self.translations.get(key, key)
    
    def change_language(self):
        """Change la langue"""
